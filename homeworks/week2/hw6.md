``` js
function isValid(arr) {
  for (let i = 0; i < arr.length; i++) {
    if (arr[i] <= 0) return 'invalid';
  }
  for (let j = 2; j < arr.length; j++) {
    if (arr[j] !== arr[j - 1] + arr[j - 2]) return 'invalid';
  }
  return 'valid';
}

isValid([3, 5, 8, 13, 22, 35]);
```
